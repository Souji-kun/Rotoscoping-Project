import bpy
import math

# --- Global Settings ---
GLOBAL_SPIN_SPEED = 10.0   # Adjust this multiplier to speed up or slow down all spins

# --- Mercury setup ---
mercury = bpy.data.objects["Mercury"]

# Clear old handlers
bpy.app.handlers.frame_change_pre.clear()

# Mercury axial tilt (~0.034°)
mercury_tilt = math.radians(0.034)

# Rotation period: ~58.65 Earth days → ~14,076 frames per spin (baseline)
mercury_spin_frames = int(240 * 58.65)

def rotate_mercury(scene):
    frame = scene.frame_current
    
    # Reset tilt each frame
    mercury.rotation_euler = (mercury_tilt, 0.0, 0.0)
    
    # Apply global spin speed multiplier
    spin_angle = (frame / (mercury_spin_frames / GLOBAL_SPIN_SPEED)) * (2 * math.pi)
    mercury.rotation_euler.rotate_axis("Z", spin_angle)

# Register handler
bpy.app.handlers.frame_change_pre.append(rotate_mercury)
