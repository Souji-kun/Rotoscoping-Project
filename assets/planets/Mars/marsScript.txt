import bpy
import math

# --- Global Settings ---
MOON_SCALE = 10.0                # Global multiplier for Phobos & Deimos size
PHOBOS_LABEL_GLOBAL_SCALE = 2.0  # Global multiplier for Phobos label size
DEIMOS_LABEL_GLOBAL_SCALE = 8.0  # Global multiplier for Deimos label size
LABEL_OFFSET = 3.0               # Vertical offset above moon (local Z)

# --- Mars setup ---
mars = bpy.data.objects["Mars"]
mars_tilt = math.radians(25.2)
mars_spin_frames = 240

# --- Phobos & Deimos setup ---
phobos = bpy.data.objects["Phobos"]
deimos = bpy.data.objects["Deimos"]

# Relative sizes (scaled by global MOON_SCALE)
mars_radius = 3389
phobos_radius = 11
deimos_radius = 6

phobos_scale = (phobos_radius / mars_radius) * MOON_SCALE
deimos_scale = (deimos_radius / mars_radius) * MOON_SCALE
phobos.scale = (phobos_scale, phobos_scale, phobos_scale)
deimos.scale = (deimos_scale, deimos_scale, deimos_scale)

# Orbital distances (compressed for clarity)
phobos_orbit = 3.0
deimos_orbit = 6.0

# Orbital periods (frames per orbit)
phobos_orbit_frames = 75
deimos_orbit_frames = 300

# --- Camera reference ---
camera = bpy.data.objects["Camera"]

# --- Create label only if not existing ---
def ensure_label(name, target_obj, scale):
    if name in bpy.data.objects:
        return bpy.data.objects[name]
    else:
        bpy.ops.object.text_add(location=(0,0,0))
        label = bpy.context.object
        label.name = name
        label.data.body = name
        label.scale = (scale, scale, scale)
        label.parent = target_obj
        
        # Add Track To constraint to face camera
        constraint = label.constraints.new(type="TRACK_TO")
        constraint.target = camera
        constraint.track_axis = 'TRACK_Z'
        constraint.up_axis = 'UP_Y'
        
        return label

phobos_label = ensure_label("Phobos_Label", phobos, PHOBOS_LABEL_GLOBAL_SCALE)
deimos_label = ensure_label("Deimos_Label", deimos, DEIMOS_LABEL_GLOBAL_SCALE)

def rotate_mars_system(scene):
    frame = scene.frame_current
    
    # --- Mars rotation ---
    mars.rotation_euler = (mars_tilt, 0.0, 0.0)
    spin_angle = (frame / mars_spin_frames) * (2 * math.pi)
    mars.rotation_euler.rotate_axis("Z", spin_angle)
    
    # --- Phobos orbit ---
    phobos_angle = (frame / phobos_orbit_frames) * (2 * math.pi)
    phobos.location = (
        mars.location.x + phobos_orbit * math.cos(phobos_angle),
        mars.location.y + phobos_orbit * math.sin(phobos_angle),
        mars.location.z
    )
    phobos.rotation_euler = (0.0, 0.0, phobos_angle)
    phobos_label.location = (0, 0, LABEL_OFFSET)
    phobos_label.scale = (PHOBOS_LABEL_GLOBAL_SCALE, PHOBOS_LABEL_GLOBAL_SCALE, PHOBOS_LABEL_GLOBAL_SCALE)
    
    # --- Deimos orbit ---
    deimos_angle = (frame / deimos_orbit_frames) * (2 * math.pi)
    deimos.location = (
        mars.location.x + deimos_orbit * math.cos(deimos_angle),
        mars.location.y + deimos_orbit * math.sin(deimos_angle),
        mars.location.z
    )
    deimos.rotation_euler = (0.0, 0.0, deimos_angle)
    deimos_label.location = (0, 0, LABEL_OFFSET)
    deimos_label.scale = (DEIMOS_LABEL_GLOBAL_SCALE, DEIMOS_LABEL_GLOBAL_SCALE, DEIMOS_LABEL_GLOBAL_SCALE)

# Clear old handlers and register
bpy.app.handlers.frame_change_pre.clear()
bpy.app.handlers.frame_change_pre.append(rotate_mars_system)
